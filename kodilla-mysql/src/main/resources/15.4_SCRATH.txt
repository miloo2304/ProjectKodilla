SELECT
    i.ID AS ISSUE_ID,
    i.SUMMARY AS TASK_SUMMARY,
    i.DESCRIPTION AS TASK_DESCRIPTION,
    il.NAME AS LIST_NAME
FROM
    ISSUES i
        JOIN
    ISSUESLISTS il ON i.ISSUESLIST_ID = il.ID;
SELECT
    i.ID AS ISSUE_ID,
    i.SUMMARY AS TASK_SUMMARY,
    i.DESCRIPTION AS TASK_DESCRIPTION,
    CONCAT(u.FIRSTNAME, ' ', u.LASTNAME) AS ASSIGNED_USER
FROM
    ISSUES i
        LEFT JOIN
    USERS u ON i.USER_ID_ASSIGNEDTO = u.ID;
SELECT
    CONCAT(u.FIRSTNAME, ' ', u.LASTNAME) AS USER_NAME,
    COUNT(i.ID) AS TASK_COUNT
FROM
    USERS u
        LEFT JOIN
    ISSUES i ON u.ID = i.USER_ID_ASSIGNEDTO
GROUP BY
    u.ID, u.FIRSTNAME, u.LASTNAME
HAVING
    TASK_COUNT > 0;
